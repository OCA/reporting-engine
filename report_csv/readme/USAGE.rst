    class PartnerCSV(models.AbstractModel):
        _name = 'report.report_csv.partner_csv'
        _inherit = 'report.report_csv.abstract'

        def generate_csv_report(self, writer, data, partners):
            writer.writeheader()
            for obj in partners:
                writer.writerow({
                    'name': obj.name,
                    'email': obj.email,
                })
